### 3.3 表单验证

```javascript
// 表单验证
class FormValidator {
  constructor(formId) {
    this.form = document.getElementById(formId);
    this.init();
  }
  
  init() {
    this.form.addEventListener('submit', (e) => {
      e.preventDefault();
      if (this.validate()) {
        this.submit();
      }
    });
    
    // 实时验证
    const inputs = this.form.querySelectorAll('input, textarea, select');
    inputs.forEach(input => {
      input.addEventListener('blur', () => {
        this.validateField(input);
      });
    });
  }
  
  validate() {
    let isValid = true;
    const inputs = this.form.querySelectorAll('[required]');
    
    inputs.forEach(input => {
      if (!this.validateField(input)) {
        isValid = false;
      }
    });
    
    return isValid;
  }
  
  validateField(input) {
    const field = input.closest('.form-field');
    const value = input.value.trim();
    let error = field.querySelector('.error-message');
    
    // 移除旧错误
    input.classList.remove('error');
    if (error) {
      error.remove();
    }
    
    // 验证
    if (input.required && !value) {
      this.showError(field, '此项为必填项');
      return false;
    }
    
    // 邮箱验证
    if (input.type === 'email' && value) {
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(value)) {
        this.showError(field, '请输入有效的邮箱地址');
        return false;
      }
    }
    
    // 手机号验证
    if (input.type === 'tel' && value) {
      const phoneRegex = /^1[3-9]\d{9}$/;
      if (!phoneRegex.test(value)) {
        this.showError(field, '请输入有效的手机号');
        return false;
      }
    }
    
    return true;
  }
  
  showError(field, message) {
    const input = field.querySelector('input, textarea, select');
    input.classList.add('error');
    
    const error = document.createElement('span');
    error.className = 'error-message';
    error.textContent = message;
    error.style.cssText = `
      color: #FF4D4F;
      font-size: 12px;
      margin-top: 8px;
      display: block;
    `;
    
    field.appendChild(error);
  }
  
  async submit() {
    const submitBtn = this.form.querySelector('button[type="submit"]');
    const originalText = submitBtn.textContent;
    
    // 设置加载状态
    submitBtn.disabled = true;
    submitBtn.innerHTML = '<span class="loading-spinner"></span> 提交中...';
    
    try {
      const formData = new FormData(this.form);
      const data = Object.fromEntries(formData);
      
      // 发送请求
      const response = await fetch('/api/submit', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data)
      });
      
      if (response.ok) {
        this.showSuccess('提交成功！');
        setTimeout(() => {
          window.location.href = '/success';
        }, 1000);
      } else {
        throw new Error('提交失败');
      }
    } catch (error) {
      this.showError('提交失败，请重试');
    } finally {
      submitBtn.disabled = false;
      submitBtn.textContent = originalText;
    }
  }
  
  showSuccess(message) {
    // 显示成功提示
    const toast = document.createElement('div');
    toast.className = 'toast toast-success';
    toast.textContent = message;
    document.body.appendChild(toast);
    
    setTimeout(() => {
      toast.remove();
    }, 2000);
  }
}

// 初始化表单
new FormValidator('createForm');
```

### 4.3 列表加载

```javascript
// 列表加载
class ListLoader {
  constructor(options) {
    this.page = 1;
    this.pageSize = options.pageSize || 20;
    this.hasMore = true;
    this.loading = false;
    this.container = document.querySelector(options.container);
    this.init();
  }
  
  init() {
    this.loadPage(1);
    
    // 滚动加载
    window.addEventListener('scroll', () => {
      if (this.shouldLoadMore()) {
        this.loadNextPage();
      }
    });
  }
  
  shouldLoadMore() {
    const scrollHeight = document.documentElement.scrollHeight;
    const scrollTop = window.scrollY;
    const clientHeight = window.innerHeight;
    
    return scrollTop + clientHeight >= scrollHeight - 100 
      && !this.loading 
      && this.hasMore;
  }
  
  async loadPage(page) {
    if (this.loading) return;
    
    this.loading = true;
    this.showLoading();
    
    try {
      const response = await fetch(`/api/list?page=${page}&size=${this.pageSize}`);
      const data = await response.json();
      
      this.renderList(data.items);
      this.hasMore = data.hasMore;
      this.page = page;
    } catch (error) {
      console.error('加载失败:', error);
      this.showError('加载失败，请重试');
    } finally {
      this.loading = false;
      this.hideLoading();
    }
  }
  
  loadNextPage() {
    if (this.hasMore) {
      this.loadPage(this.page + 1);
    }
  }
  
  renderList(items) {
    items.forEach(item => {
      const element = this.createListItem(item);
      this.container.appendChild(element);
    });
  }
  
  createListItem(item) {
    const div = document.createElement('div');
    div.className = 'list-item';
    div.innerHTML = `
      <div class="item-icon">${item.icon}</div>
      <div class="item-content">
        <h3 class="item-title">${item.title}</h3>
        <p class="item-desc">${item.description}</p>
        <div class="item-meta">
          <span class="item-time">${item.time}</span>
          <span class="item-status status-${item.status}">${item.statusText}</span>
        </div>
      </div>
      <div class="item-action">
        <button class="btn-icon">›</button>
      </div>
    `;
    
    div.addEventListener('click', () => {
      window.location.href = `/detail/${item.id}`;
    });
    
    return div;
  }
  
  showLoading() {
    // 显示加载指示器
  }
  
  hideLoading() {
```
